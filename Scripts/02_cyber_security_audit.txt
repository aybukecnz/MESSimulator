CREATE TABLE SystemAuditLogs (
    LogId SERIAL PRIMARY KEY,
    Timestamp TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    IpAddress VARCHAR(45) NOT NULL,
    MacAddress VARCHAR(17) NOT NULL,
    TargetTable VARCHAR(50) NOT NULL,
    ActionType VARCHAR(20) NOT NULL, -- 'INSERT', 'UPDATE', 'SPOOFING_ATTEMPT' vb.
    Message TEXT NOT NULL
);
CREATE TABLE MachineRecipes (
    RecipeId SERIAL PRIMARY KEY,
    ParameterName VARCHAR(50) NOT NULL,
    Value NUMERIC(10,4) NOT NULL,
    UpdatedAt TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Reçete Deđiţimini Ýzleyen Siber Fonksiyon
CREATE OR REPLACE FUNCTION track_recipe_manipulation()
RETURNS TRIGGER AS $$
BEGIN
    -- Birisi kritik parametreyi deđiţtirmeye çalýţýrsa siber audit logu oluţtur
    INSERT INTO SystemAuditLogs (IpAddress, MacAddress, TargetTable, ActionType, Message)
    VALUES (
        '192.168.1.150', -- Normalde backendden veya sessiondan alýnýr, ţimdilik statik simüle ediyoruz
        '00:1A:2B:3C:4D:5E', 
        'MachineRecipes', 
        'CRITICAL_UPDATE', 
        'Kritik üretim reçetesi deđiţtirildi! Eski Deđer: ' || OLD.Value || ' -> Yeni Deđer: ' || NEW.Value
    );
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

-- Trigger'ý reçete tablosuna bađlama
CREATE TRIGGER trg_track_recipe_changes
AFTER UPDATE OR DELETE ON MachineRecipes
FOR EACH ROW
EXECUTE FUNCTION track_recipe_manipulation();
