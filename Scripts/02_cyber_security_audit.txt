-- 1. KRÝTÝK ÜRETÝM REÇETELERÝ TABLOSU
-- Fabrikadaki jeneratör, motor ve türbin ayar parametrelerini tutar.
CREATE TABLE IF NOT EXISTS public.MachineRecipes (
    RecipeId SERIAL PRIMARY KEY,
    ParameterName VARCHAR(50) NOT NULL,
    Value NUMERIC(10,4) NOT NULL,
    UpdatedAt TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 2. BAŢLANGIÇ SÝMÜLASYON VERÝSÝ (SEED DATA)
-- Eđer parametre tabloda yoksa, test verisini güvenli bir ţekilde ekler.
INSERT INTO public.MachineRecipes (ParameterName, Value)
SELECT 'TargetRpmLimit', 1500.0
WHERE NOT EXISTS (
    SELECT 1 FROM public.MachineRecipes WHERE ParameterName = 'TargetRpmLimit'
);

-- 3. SÝBER MANÝPÜLASYON ÝZLEME FONKSÝYONU (PL/pgSQL)
-- Reçete tablosunda bir deđiţiklik olduđunda .NET'i beklemeden dođrudan çalýţýr.
CREATE OR REPLACE FUNCTION track_recipe_manipulation()
RETURNS TRIGGER AS $$
BEGIN
    INSERT INTO public.systemauditlogs (timestamp, sourceip, username, actiontype, status, details)
    VALUES (
        NOW(),                  -- Veritabanýnýn zorunlu tuttuđu zaman damgasý
        '192.168.1.150',        -- Simüle edilen ţüpheli IP adresi
        'Hacker_Anonymous',     -- Yetkisiz iţlem gerçekleţtiren sahte aktör
        'RECIPE_MANIPULATION',  -- Siber güvenlik olay tipi
        'CRITICAL',             -- Olayýn kritiklik seviyesi
        'Kritik üretim reçetesi izinsiz deđiţtirildi! Eski Deđer: ' || OLD.Value || ' -> Yeni Deđer: ' || NEW.Value
    );
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

-- 4. TETÝKLEYÝCÝ (TRIGGER) KURULUMU
-- Fonksiyonu MachineRecipes tablosunun UPDATE ve DELETE olaylarýna bađlar.
DROP TRIGGER IF EXISTS trg_track_recipe_changes ON public.MachineRecipes;

CREATE TRIGGER trg_track_recipe_changes
    AFTER UPDATE OR DELETE ON public.MachineRecipes
    FOR EACH ROW
    EXECUTE FUNCTION track_recipe_manipulation();
